一 JavaScript概述

作用：用户和页面交互效果

特点：js是一个基于对象弱类型的脚本语言，浏览器直接解释运行的

组成：ECMAScript、BOM、DOM

1.1 JavaScript作用

技术 作用
HTML 用于页面的搭建
CSS 用于页面的美化
javaScript 用户与页面的交互

1.2 JavaScript的特点

js是一个弱类型的 脚本语言，浏览器直接解析运行

特点 Java javaScript
面向对象 完全面向对象的语言：继承、封装、多态 基于对象的语言，不完全符合面向对象的思想
运行方式 编译型，运行过程需要生成字节码文件 解释型语言，不会生成中间文件，解释一定行数，再执行。
跨平台 安装了JVM就可以运行在不同的操作系统中 只要有浏览器的地方就可以运行
大小写 区分大小写 区分大小写
数据类型 强类型语言，不同的数据类型有严格区分 弱类型语言，不同类型的数据可以直接赋值给同一个变量。

1.3 JavaScript的语法组成

组成部分 作用
ECMA Script 构成了JS核心的语法基础
BOM Browser Object Model 浏览器对象模型，用来操作浏览器上的对象
DOM Document Object Model 文档对象模型，用来操作网页中的 元素（标签）

二 JavaScript基础语法

2.1 html与js结合方式

<!DOCTYPE html>
<html lang="zh-CN">
<head>
    <meta charset="UTF-8">
    <title>01_html与js结合方式</title>
    <!--
        type="text/javascript"（默认），当浏览器加载到此标签，解析器改为js
    -->
    <script type="text/javascript">
        // js代码   文档 document对象
        document.write('<h3>你好，传智播客</h3>');
    </script>
    <script src="../js/myjs.js">
        document.write('<h3>我是外部标签引入，又在标签体内编写js代码</h3>');
    </script>
</head>
<body>
<!--
    html与js结合方式
        1）内部脚本
            格式：使用script标签，在标签体内编写js代码
        2）外部脚本
            格式：使用script标签，在src属性中引入外部的js文件


        注意：
            1）script标签可以在页面的任意位置，推荐放在body下方 （浏览器加载顺序 自上而下）
                在企业开发中 一般情况下 css在head，js在body下方
            2）script标签可以出现多次，浏览器依次加载解析运行
            3）如果使用script标签中的src属性，那么标签体内的js代码不再被浏览器加载解析
-->
</body>
</html>

2.2 js变量的定义

java是强类型语言，注重变量的定义，所以在java中定义变量类型的方式如下

// 整型
int i = 1314;
// 浮点型
double d = 521.1314;
// 字符型
char c ='c';
// 字符串型
String str = "用心做教育";
//布尔型
boolean b = true;

js是弱类型语言，不注重变量的定义，所以在js中定义变量的类型方式如下

<script>
    // 整型
    var i = 1234;  // es6 提供了块级作用域
    // 浮点型
    var d = 123.321;
    // 字符型
    // 补充知识点：在js中，没有字符只有字符串 单引号和双引号都可以
    var c = 'ccccc';
    // 字符串
    var str = "传智播客，用心做教育...";
    // 布尔型
    // 我们在定义变量时可以省略 var （默认）
    b = true;

</script>

2.3 js数据类型

js与java一样，数据类型也分为基本数据类型（原始数据类型）和引用数据类型

a）原始数据类型（5种）

number：数值类型（整型和浮点型）
string：字符串型（字符和字符串）
boolean：布尔型
null：空值 例如：var a = null; // 引用数据类型的占位符  Object
undefined：未定义 例如：var a;

b）引用数据类型

js与java一样，使用new关键字来创建一个对象

// 上帝对象
var obj = new Object();
// 日期对象
var date = new Date();

c）typeof操作符

作用：判断变量的数据类型

格式：typeof 变量名 或者 typeof(变量名)；

<script>
    // 整型
    var i = 1234;
    // 在java中  sout=System.out.println  （控制台）-----> console.log(.....); （控制台）
    // 在js中，一行代码结束可以省略分号，不建议
    console.log("整型："+typeof (i))
    // 浮点型
    var d = 123.321;
    console.log("浮点型："+typeof (d));
    // 字符型
    // 补充知识点：在js中，没有字符只有字符串 单引号和双引号都可以
    var c = 'ccccc';
    console.log("字符型："+typeof (c));
    // 字符串
    var str = "传智播客，用心做教育...";
    console.log("字符串："+typeof (str));
    // 布尔型
    // 我们在定义变量时可以省略 var （默认）
    b = true;
    console.log("布尔型：" + typeof b);
    // 空值
    var n = null;
    console.log("空值：" + typeof n);
    // 未定义
    var u;
    console.log("未定义：" + typeof u);
    // 引用数据类型
    // 在js种，使用var声明的变量，可以多次定义
    var u = new Date();
    console.log("引用数据类型：" + typeof u);

</script>

2.4 js运算符

js与java的运算符一样，什么算数运算符、赋值运算符、逻辑运算符等等，不用把这些运算符都记住，写程序时自然会用就OK。

a）算术运算符 ：+ - * / % ++ --

b）赋值运算符：= += -= *= /=

c）比较运算符：> < == === != !==

d）逻辑运算符：&& || !

e）三目运算符：条件表达式？为真:为假

下面说几个与java不一样的js运算符：

<script>
    // 算术运算符
    // 在js中，字符串可以和数值进行数学运算，底层实现了转换技术
    var a = 10;
    var b = '3';
    console.log(a + b);// 字符串拼接
    console.log(a - b);
    console.log(a * b);
    console.log(a / b);//不再是整除，保留小数位 （后期数据存储的价格 单位是分）
    // += *=

    // 比较运算符
    var c = "10";
    console.log(a == c); // 返回结果 true 比较内容
    console.log(a === c);// 返回结果 false  全等运算符（恒等） 比较的类型和内容

    // != !==
    console.log(a != c); // 返回结果 false
    console.log(a !== c);// 返回结果  true
</script>

2.5 流程控制语句

在高级语言中：顺序、分支、循环

a）if判断

if(条件表达式){
    // js代码
}else if(条件表达式){
    // js代码
}else{
    // js代码
}
<script>
    // 条件表达式
    /*
    *   数值类型：非0为真
    *   字符串类型：非空串为真
    *   变量： null 和 undefined 都为假
    *
    * */
    //  var flag = false;
    // flag = 0;
    // flag = " ";
    // flag =null;
    flag = undefined;
    // if判断
    if (flag) {
        document.write('<h3>为真</h3>');
    } else {
        document.write('<h3>为假</h3>');
    }
</script>

b）循环

// for
for(var i =0;i < length; i ++){
    // 循环体
}
// while
while(条件表达式){
    // 循环体
}
// do..while
do{
    // 循环体
    // continue; 跳出本次循环
    // break;跳出整个循环
}while(条件表达式);
<script>
    // for
    for (var i = 0; i < 5; i++) {
        document.write(i + "<br/>");
        //document.writeln(i);
    }
    // 拓展知识点；使用var声明的变量 全局变量
    console.log(i);
    // 在js 6的版本中，提供 另外一种变量的let声明  （块级作用域）  这不是局部变量
    for (let j = 0; j < 5; j++) {
        document.write(j + '<br/>');
    }
    // j is not defined
    console.log(j);
</script>

2.8 函数（方法）使用

a）普通函数

// 格式
function 函数名(参数列表){
    // js代码
    [return 返回值;]
}
<script>
    // 普通函数

    // 注意：在js中没有方法重载，重名的方法会被覆盖
    // 再来一个3个数相加的和
    function sum(a, b, c) {
        return a + b + c;
    }
    // 来一个2个数相加的和
    function sum(a, b) {
        // 在js函数中，内部有一个arguments(数组)对象，获取实际的参数列表
        console.log(arguments);
        console.log(arguments[0]);
        console.log(arguments[1]);
        console.log(arguments[2]);
        console.log(arguments[3]);
        return a + b;
    }

    // NaN not a number 这不是一个数
    // 不会报错，弱类型语言  我们在调用方法传递的实际参数个数与定义方式无关
    console.log("返回结果："+sum(5,10,15,20,25,30));
</script>

b）匿名函数

// 格式   一会我会讲js事件                  // 匿名函数，通常与事件结合使用
var fun = function (参数列表){
    [return 返回值;]
}
<script>
    // 匿名函数
    var fun = function (name) {
        console.log(name + "喜欢购物");
    }

    fun('陈浩');
</script>          

<input id="myBtn" type="button" value="" />
<script>
    document.getElementById('myBtn').onclick=function(){
        // js代码
    }
</script>

c）变量作用域

<!--
    变量作用域：
        全局变量
            特点：作用域范围在整个页面
        块级变量
            特点：在我们局部范围 （要求浏览器的版本高）
        局部变量
            特点：在函数中，函数执行完毕变量销毁
            特殊：在函数中，如果未使用var声明变量，此变量默认为全局变量
-->

<script>
   var a = 5;

    function show() {
        var b = 10;
        console.log("我在函数内输出的局部变量b：" + b);
        c = 15;
        console.log("我在函数内输出的未声明c：" + c);
    } 

    show();
    // b is not defined
    // console.log("我在函数外输出的局部变量b：" + b);
    console.log("我在函数外输出的未声明c：" + c);
</script>
<script>
    console.log("我是全局变量:" + a);
</script>

三 JavaScript事件

3.1 作用

js通过事件监听用户的行为，结合函数完成页面的交互效果

3.2 常见事件

属性 描述
onblur 元素失去焦点
onfocus 元素获得焦点
onchange 用户改变域的内容
onclick 鼠标点击某个对象
ondblclick 鼠标双击某个对象
onkeydown 某个键盘的键被按下
onkeyup 某个键盘的键被松开
onload 某个页面或图像被完成加载
onmousedown 某个鼠标按键被按下
onmouseout 鼠标从某元素移开
onmouseover 鼠标被移到某元素之上
onmouseup 某个鼠标按键被松开
onsubmit 提交按钮被点击

3.3 事件注册方式

标签绑定事件，完成监听的效果

<body>
<input type="button" value="我是普通函数" onclick="sing()"> <br>
<input type="button" value="我是匿名函数" id="myBtn">
</body>
<script>
    /*
    *   事件绑定方式有二种：
    *       普通函数（耦合性太高）
    *       匿名函数（解耦）  ---- 有2天来学习 jquery（jdbc-jdbTemplate）
    *       
    *       用单击事件举例 onClick
    * 
    * */
    function sing() {
        window.alert('单身情歌');
    }

    // 获取按钮对象
    document.getElementById("myBtn").onclick = function () {
        window.alert('海草舞');
    }
</script>

四 JavaScript内置对象

4.1 数组对象

<script>
    /*
    * 数组对象
    *   1） var arr = [元素1，元素2，元素3....];（掌握）
    *   2） var arr = new Array();
    *   3） var arr = new Array(长度);
    *   4） var arr = new Array(元素1，元素2，元素3....);
    *
    *
    *
    *   注意：在js中，数组的元素类型和长度任意,很像java中的集合
    *
    * */

    var arr = ['鹏哥', "牛哥", "婷哥", "擎哥"];
    console.log(arr);

    var arr2 = new Array();
    arr2[0] = "苍老师";
    arr2[1] = "波多老师";
    arr2[2] = "小泽老师";
    console.log(arr2);

    var arr3 = new Array(2);
    arr3[0] = "花泽老师";
    arr3[1] = "大桥老师";
    arr3[2] = 1234;
    arr3[3] = new Date();
    arr3[4] = true;
    console.log(arr3);

    var arr4 = new Array(1234, true, '张三');
    console.log(arr4);
    /*
    *   数组常见方法
    *       1）concat() 连接数组
    *       2）join('拼接规则') 与split功能相反，将数组元素拼接成一个字符串
    *       3）reverse() 数组元素反转
    *       4）sort() 排序
    *           字符串 按照字母顺序
    *           数值
    *               sort(function(a,b){
    *                   return a-b; 升序
    *                   reutnr b-a; 降序
    *               });
    *
    *
    * */

    console.log(arr.concat(arr2));
    console.log(arr2.join("+"));
    console.log(arr3.reverse());

    var arr5 = ["c", "b", "a"];
    console.log(arr5.sort());
    var arr6 = [11, 9, 1, 8, 5];
    console.log(arr6.sort(function (a, b) {
        return a - b;
    }));
</script>

4.2 日期对象

<script>
    // 日期对象
    var date = new Date();
    console.log(date);
    console.log(date.toLocaleString()); // 根据系统的本地时间来显示
    console.log(date.getFullYear());// yyyy
    console.log(date.getMonth());//MM
    console.log(date.getDate());// dd
    console.log(date.getDay());// 星期
    console.log(date.getTime());// 获取 1970至今的毫秒值

</script>

4.3 全局对象

不需要创建直接使用

<script>
    // 字符串转换数值
    /*
    *   parseInt(); 字符串转换成整型
    *   parseFloat(); 字符串转换成浮点型
    *       特点：从字符串第一个字符开始转换，当遇到非数值字符串停止转换
    *       应用场景：
    *               图片修改大小 100px 200px
    *               获取鼠标左表(x,y) onmousemove
    *
    *       NaN  not a number  这不是一个数
    *
    *       isNaN(); 判断这个字符串不是数值 返回true
    *
    * */

    var a = "a1a23.a3a21";
    var b = "123";
    console.log(parseInt(a));

    console.log(parseFloat(a));

    console.log(isNaN(a));
    console.log(isNaN(b));


    // 编码和解码
    /*
    *   浏览器发送数据到服务器   中文---编码
    *       encodeURI();
    *   服务器返回数据到浏览器 中文--解码
    *       decodeURI();
    * */

    var str = "鹏哥年轻有为，活力棒棒哒.....";
    // 编码
    var encodeURI = encodeURI(str);
    console.log(encodeURI);
    // 解码
    console.log(decodeURI(encodeURI));

</script>

JavaScript二维数组

其实，我们都见过这样的二维数组，只不过平时没在意罢了，例如：

//这就是一个二位数组。角标从0开始，这样看起来比较繁琐
var arr = [[1,2,4,6],[2,4,7,8],[8,9,10,11],[9,12,13,15]] 
//一般我们会这样定义二维数组
var arr = new Array();
arr[0]=[1,2,4,6];
arr[1]=[2,4,7,8];
arr[2]=[8,9,10,11];
arr[3]=[9,12,13,15];

例题：

var arr = [[1,2],['a','b']];
console.log(arr[1]) // 输出结果为['a','b']
console.log(arr[1][0]); //输出结果为a，第2列第1行所在的元素

var arr = new Array(new Array(1,2),new Array("a","b"));
console.log(arr[1]) // 输出结果为['a','b']
console.log(arr[1][0]); //输出结果为a，第2列第1行所在的元素
